## 我在做什么？

上一章的回显服务器有一个致命缺陷：**一次只能服务一个客户端**。第二个客户端连接时，必须等第一个断开后才能被处理。这就像一家只有一个服务员的餐厅——顾客必须排队，前面的不走，后面的永远吃不上。

真实的服务器（如 Nginx、Apache、数据库）必须能同时处理成千上万的连接。实现方式有三种：
1. **多进程**：每个客户端 fork 一个子进程
2. **多线程**：每个客户端创建一个线程
3. **IO多路复用**：一个线程管理多个连接（最主流）

本章依次讲解这三种方式，最后实现一个**多客户端聊天服务器**。

`*此术语见附录F*`：IO多路复用、select、epoll。

## 56.1 单线程服务器的问题

回顾上一章的回显服务器结构：

```
单线程服务器:
  ┌─────────────────────────────────────┐
  │  while(1) {                         │
  │    client = accept();  ← 阻塞等待    │
  │    while(1) {                       │
  │      recv(client);     ← 阻塞等待    │
  │      send(client);                  │
  │    }                                │
  │  }                                  │
  └─────────────────────────────────────┘

客户端1 连接后，服务端进入 recv 循环
客户端2 尝试连接，但 accept() 永远不会被调用
→ 客户端2 永远等不到服务！
```

问题根源：`accept()` 和 `recv()` 都是阻塞调用，一个客户端占用了整个服务端。

## 56.2 方案一：多进程并发

每接受一个连接，`fork()` 一个子进程专门处理该客户端：

```
多进程服务器:
                    ┌──────────┐
                    │ 父进程    │
                    │ accept() │
                    └────┬─────┘
                         │ 新连接
               ┌─────────┼─────────┐
               │ fork()  │         │ fork()
          ┌────▼───┐ ┌───▼────┐ ┌──▼──────┐
          │ 子进程1 │ │ 子进程2 │ │ 子进程3  │
          │ recv/send│ │ recv/send│ │ recv/send│
          │ 客户端1  │ │ 客户端2  │ │ 客户端3  │
          └────────┘ └────────┘ └─────────┘
```

```c
/*
 * multi_process_server.c —— 多进程并发服务器
 * 编译: gcc -std=c99 -Wall -o mp_server multi_process_server.c
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <signal.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <sys/wait.h>
#include <netinet/in.h>
#include <arpa/inet.h>

#define PORT 8080
#define BUFFER_SIZE 1024

/* 信号处理函数：回收僵尸子进程 */
void sigchld_handler(int sig) {
    /* waitpid(-1, ...) 回收任意子进程，WNOHANG 不阻塞 */
    while (waitpid(-1, NULL, WNOHANG) > 0);
}

void handle_client(int client_fd, struct sockaddr_in client_addr) {
    char client_ip[INET_ADDRSTRLEN];
    inet_ntop(AF_INET, &client_addr.sin_addr, client_ip, sizeof(client_ip));
    printf("[子进程 %d] 客户端 %s:%d 已连接\n",
           getpid(), client_ip, ntohs(client_addr.sin_port));

    char buffer[BUFFER_SIZE];
    while (1) {
        memset(buffer, 0, BUFFER_SIZE);
        ssize_t bytes = recv(client_fd, buffer, BUFFER_SIZE - 1, 0);

        if (bytes <= 0) {
            if (bytes == 0) {
                printf("[子进程 %d] 客户端断开\n", getpid());
            } else {
                perror("recv 失败");
            }
            break;
        }

        buffer[bytes] = '\0';
        printf("[子进程 %d] 收到: %s", getpid(), buffer);

        /* 回显 */
        if (send(client_fd, buffer, bytes, 0) < 0) {
            perror("send 失败");
            break;
        }
    }

    close(client_fd);
    printf("[子进程 %d] 结束\n", getpid());
    exit(0);
}

int main(void) {
    int server_fd;
    struct sockaddr_in server_addr;

    /* 注册 SIGCHLD 处理函数，避免僵尸进程 */
    signal(SIGCHLD, sigchld_handler);

    /* 创建 socket */
    server_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (server_fd < 0) {
        perror("socket 失败");
        exit(1);
    }

    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    /* 绑定 */
    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(PORT);

    if (bind(server_fd, (struct sockaddr*)&server_addr,
             sizeof(server_addr)) < 0) {
        perror("bind 失败");
        exit(1);
    }

    /* 监听 */
    if (listen(server_fd, 10) < 0) {
        perror("listen 失败");
        exit(1);
    }

    printf("多进程服务器启动，端口 %d\n", PORT);

    /* 主循环：接受连接，fork 子进程 */
    while (1) {
        struct sockaddr_in client_addr;
        socklen_t client_len = sizeof(client_addr);

        int client_fd = accept(server_fd,
                               (struct sockaddr*)&client_addr,
                               &client_len);
        if (client_fd < 0) {
            perror("accept 失败");
            continue;
        }

        pid_t pid = fork();
        if (pid < 0) {
            perror("fork 失败");
            close(client_fd);
            continue;
        } else if (pid == 0) {
            /* 子进程：不需要 server_fd */
            close(server_fd);
            handle_client(client_fd, client_addr);
            /* handle_client 内部 exit(0)，不会执行到这里 */
        } else {
            /* 父进程：不需要 client_fd */
            close(client_fd);
        }
    }

    close(server_fd);
    return 0;
}
```

**优点**：简单，进程隔离好，一个子进程崩溃不影响其他。
**缺点**：每个连接一个进程，开销大，不适合高并发（C10K 问题）。

## 56.3 方案二：多线程并发

每接受一个连接，创建一个线程处理：

```c
/*
 * multi_thread_server.c —— 多线程并发服务器
 * 编译: gcc -std=c99 -Wall -pthread -o mt_server multi_thread_server.c
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <pthread.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>

#define PORT 8080
#define BUFFER_SIZE 1024

typedef struct {
    int client_fd;
    struct sockaddr_in addr;
} ClientInfo;

void* handle_client(void *arg) {
    ClientInfo *info = (ClientInfo*)arg;
    int client_fd = info->client_fd;

    char client_ip[INET_ADDRSTRLEN];
    inet_ntop(AF_INET, &info->addr.sin_addr, client_ip, sizeof(client_ip));
    printf("[线程 %lu] 客户端 %s:%d 已连接\n",
           pthread_self(), client_ip, ntohs(info->addr.sin_port));

    free(info);  /* 释放动态分配的 ClientInfo */

    char buffer[BUFFER_SIZE];
    while (1) {
        memset(buffer, 0, BUFFER_SIZE);
        ssize_t bytes = recv(client_fd, buffer, BUFFER_SIZE - 1, 0);

        if (bytes <= 0) {
            if (bytes == 0) {
                printf("[线程 %lu] 客户端断开\n", pthread_self());
            }
            break;
        }

        buffer[bytes] = '\0';
        printf("[线程 %lu] 收到: %s", pthread_self(), buffer);

        if (send(client_fd, buffer, bytes, 0) < 0) {
            break;
        }
    }

    close(client_fd);
    return NULL;
}

int main(void) {
    int server_fd;
    struct sockaddr_in server_addr;

    server_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (server_fd < 0) {
        perror("socket 失败");
        exit(1);
    }

    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(PORT);

    if (bind(server_fd, (struct sockaddr*)&server_addr,
             sizeof(server_addr)) < 0) {
        perror("bind 失败");
        exit(1);
    }

    if (listen(server_fd, 10) < 0) {
        perror("listen 失败");
        exit(1);
    }

    printf("多线程服务器启动，端口 %d\n", PORT);

    while (1) {
        ClientInfo *info = (ClientInfo*)malloc(sizeof(ClientInfo));
        if (info == NULL) {
            perror("malloc 失败");
            continue;
        }

        socklen_t addr_len = sizeof(info->addr);
        info->client_fd = accept(server_fd,
                                 (struct sockaddr*)&info->addr,
                                 &addr_len);
        if (info->client_fd < 0) {
            perror("accept 失败");
            free(info);
            continue;
        }

        pthread_t thread;
        pthread_create(&thread, NULL, handle_client, info);
        pthread_detach(thread);  /* 分离线程，自动回收资源 */
    }

    close(server_fd);
    return 0;
}
```

**`pthread_detach`**：分离线程后，线程结束时自动回收资源，不需要 `pthread_join`。

**优点**：比多进程轻量，创建和切换开销小。
**缺点**：线程共享地址空间，需要处理竞态条件；线程数过多时上下文切换开销大。

## 56.4 方案三：IO多路复用（select）

多进程和多线程都是"一个连接一个执行流"。IO多路复用的思想是：**一个线程同时监控多个文件描述符，哪个有数据就处理哪个**。

```
IO多路复用:
  ┌───────────────────────────────────────┐
  │  主线程                                │
  │                                       │
  │  fd_set 监控:                          │
  │  ┌────┬────┬────┬────┬────┐          │
  │  │ fd │ fd │ fd │ fd │ fd │  ...     │
  │  │ 3  │ 4  │ 5  │ 6  │ 7  │          │
  │  └────┴────┴────┴────┴────┘          │
  │    │    │    │         │               │
  │    ▼    ▼    ▼         ▼               │
  │  select() 返回哪些 fd 就绪             │
  │  然后逐一处理就绪的 fd                  │
  └───────────────────────────────────────┘
```

### select 模型

```c
/*
 * select_server.c —— select IO多路复用服务器
 * 编译: gcc -std=c99 -Wall -o select_server select_server.c
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <sys/select.h>
#include <netinet/in.h>
#include <arpa/inet.h>

#define PORT 8080
#define BUFFER_SIZE 1024
#define MAX_CLIENTS 100

int main(void) {
    int server_fd, client_fds[MAX_CLIENTS];
    struct sockaddr_in server_addr;
    fd_set read_fds;  /* 监控可读的文件描述符集合 */
    int max_fd;

    /* 初始化 client_fds 数组 */
    for (int i = 0; i < MAX_CLIENTS; i++) {
        client_fds[i] = -1;
    }

    /* 创建 socket */
    server_fd = socket(AF_INET, SOCK_STREAM, 0);
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(PORT);

    bind(server_fd, (struct sockaddr*)&server_addr, sizeof(server_addr));
    listen(server_fd, 10);

    printf("select 多路复用服务器启动，端口 %d\n", PORT);

    while (1) {
        /* 清空并重建 fd_set */
        FD_ZERO(&read_fds);
        FD_SET(server_fd, &read_fds);
        max_fd = server_fd;

        /* 把所有客户端 fd 加入集合 */
        for (int i = 0; i < MAX_CLIENTS; i++) {
            if (client_fds[i] > 0) {
                FD_SET(client_fds[i], &read_fds);
                if (client_fds[i] > max_fd) {
                    max_fd = client_fds[i];
                }
            }
        }

        /* select 阻塞等待任意 fd 就绪 */
        int activity = select(max_fd + 1, &read_fds, NULL, NULL, NULL);
        if (activity < 0) {
            perror("select 失败");
            continue;
        }

        /* 1. 检查是否有新连接 */
        if (FD_ISSET(server_fd, &read_fds)) {
            struct sockaddr_in client_addr;
            socklen_t addr_len = sizeof(client_addr);
            int new_fd = accept(server_fd,
                                (struct sockaddr*)&client_addr,
                                &addr_len);
            if (new_fd >= 0) {
                /* 找空位存放新客户端 */
                for (int i = 0; i < MAX_CLIENTS; i++) {
                    if (client_fds[i] == -1) {
                        client_fds[i] = new_fd;
                        char ip[INET_ADDRSTRLEN];
                        inet_ntop(AF_INET, &client_addr.sin_addr,
                                  ip, sizeof(ip));
                        printf("新客户端 [%d]: %s:%d\n",
                               i, ip, ntohs(client_addr.sin_port));
                        break;
                    }
                }
            }
        }

        /* 2. 检查哪些客户端有数据到达 */
        for (int i = 0; i < MAX_CLIENTS; i++) {
            int fd = client_fds[i];
            if (fd > 0 && FD_ISSET(fd, &read_fds)) {
                char buffer[BUFFER_SIZE];
                memset(buffer, 0, BUFFER_SIZE);
                ssize_t bytes = recv(fd, buffer, BUFFER_SIZE - 1, 0);

                if (bytes <= 0) {
                    /* 客户端断开 */
                    printf("客户端 [%d] 断开\n", i);
                    close(fd);
                    client_fds[i] = -1;
                } else {
                    buffer[bytes] = '\0';
                    printf("客户端 [%d] 发来 %zd 字节: %s", i, bytes, buffer);

                    /* 回显 */
                    send(fd, buffer, bytes, 0);
                }
            }
        }
    }

    close(server_fd);
    return 0;
}
```

### select 的局限

- `FD_SETSIZE` 通常为 1024，最多监控 1024 个 fd
- 每次调用都要重新设置 `fd_set`，O(n) 复杂度
- 每次都要遍历所有 fd 检查哪些就绪

## 56.5 epoll：Linux 高性能 IO 多路复用

epoll 是 Linux 特有的 IO 多路复用机制，解决了 select 的性能问题：

```
select vs epoll:

select:                           epoll:
  ┌──────────────────┐            ┌──────────────────┐
  │ 每次调用传入所有fd │            │ 只需注册一次      │
  │ 内核遍历所有fd    │            │ 内核维护就绪列表  │
  │ 返回后用户遍历所有fd│           │ 只返回就绪的fd    │
  │ O(n) 复杂度      │            │ O(1) 获取就绪事件 │
  └──────────────────┘            └──────────────────┘
```

```c
/*
 * epoll_server.c —— epoll IO多路复用服务器
 * 编译: gcc -std=c99 -Wall -o epoll_server epoll_server.c
 * 注意: 仅 Linux 支持 epoll
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <sys/epoll.h>
#include <netinet/in.h>
#include <arpa/inet.h>

#define PORT 8080
#define BUFFER_SIZE 1024
#define MAX_EVENTS 100

int main(void) {
    int server_fd, epoll_fd;
    struct sockaddr_in server_addr;
    struct epoll_event ev, events[MAX_EVENTS];

    /* 创建 socket */
    server_fd = socket(AF_INET, SOCK_STREAM, 0);
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(PORT);

    bind(server_fd, (struct sockaddr*)&server_addr, sizeof(server_addr));
    listen(server_fd, 10);

    /* 创建 epoll 实例 */
    epoll_fd = epoll_create1(0);
    if (epoll_fd < 0) {
        perror("epoll_create1 失败");
        exit(1);
    }

    /* 将 server_fd 加入 epoll 监控 */
    ev.events = EPOLLIN;  /* 监控可读事件 */
    ev.data.fd = server_fd;
    if (epoll_ctl(epoll_fd, EPOLL_CTL_ADD, server_fd, &ev) < 0) {
        perror("epoll_ctl 失败");
        exit(1);
    }

    printf("epoll 服务器启动，端口 %d\n", PORT);

    while (1) {
        /* 等待事件，-1 表示无限等待 */
        int nfds = epoll_wait(epoll_fd, events, MAX_EVENTS, -1);
        if (nfds < 0) {
            perror("epoll_wait 失败");
            continue;
        }

        for (int i = 0; i < nfds; i++) {
            if (events[i].data.fd == server_fd) {
                /* 新连接 */
                struct sockaddr_in client_addr;
                socklen_t addr_len = sizeof(client_addr);
                int client_fd = accept(server_fd,
                                       (struct sockaddr*)&client_addr,
                                       &addr_len);
                if (client_fd >= 0) {
                    char ip[INET_ADDRSTRLEN];
                    inet_ntop(AF_INET, &client_addr.sin_addr,
                              ip, sizeof(ip));
                    printf("新客户端: %s:%d\n",
                           ip, ntohs(client_addr.sin_port));

                    /* 将客户端 fd 加入 epoll */
                    ev.events = EPOLLIN;
                    ev.data.fd = client_fd;
                    epoll_ctl(epoll_fd, EPOLL_CTL_ADD, client_fd, &ev);
                }
            } else {
                /* 客户端数据 */
                int client_fd = events[i].data.fd;
                char buffer[BUFFER_SIZE];
                memset(buffer, 0, BUFFER_SIZE);
                ssize_t bytes = recv(client_fd, buffer, BUFFER_SIZE - 1, 0);

                if (bytes <= 0) {
                    /* 客户端断开 */
                    printf("客户端 %d 断开\n", client_fd);
                    epoll_ctl(epoll_fd, EPOLL_CTL_DEL, client_fd, NULL);
                    close(client_fd);
                } else {
                    buffer[bytes] = '\0';
                    printf("收到 %zd 字节: %s", bytes, buffer);
                    send(client_fd, buffer, bytes, 0);
                }
            }
        }
    }

    close(server_fd);
    close(epoll_fd);
    return 0;
}
```

### epoll 关键函数

| 函数 | 说明 |
|------|------|
| `epoll_create1(0)` | 创建 epoll 实例 |
| `epoll_ctl(epfd, op, fd, event)` | 添加/修改/删除监控的 fd |
| `epoll_wait(epfd, events, max, timeout)` | 等待事件发生 |
| `EPOLL_CTL_ADD` | 添加 fd 到监控列表 |
| `EPOLL_CTL_MOD` | 修改监控事件 |
| `EPOLL_CTL_DEL` | 从监控列表删除 fd |
| `EPOLLIN` | 可读事件 |
| `EPOLLOUT` | 可写事件 |

## 56.6 简单聊天服务器

下面是一个多线程聊天服务器，所有客户端发来的消息会广播给其他所有客户端：

```c
/*
 * chat_server.c —— 简单聊天服务器
 * 编译: gcc -std=c99 -Wall -pthread -o chat_server chat_server.c
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <pthread.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>

#define PORT 8080
#define BUFFER_SIZE 1024
#define MAX_CLIENTS 50

/* 全局客户端列表（受互斥锁保护） */
int client_fds[MAX_CLIENTS];
int client_count = 0;
pthread_mutex_t clients_mutex = PTHREAD_MUTEX_INITIALIZER;

/* 广播消息给所有客户端 */
void broadcast(const char *message, int sender_fd) {
    pthread_mutex_lock(&clients_mutex);
    for (int i = 0; i < MAX_CLIENTS; i++) {
        int fd = client_fds[i];
        if (fd > 0 && fd != sender_fd) {
            send(fd, message, strlen(message), 0);
        }
    }
    pthread_mutex_unlock(&clients_mutex);
}

/* 添加客户端 */
void add_client(int fd) {
    pthread_mutex_lock(&clients_mutex);
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (client_fds[i] <= 0) {
            client_fds[i] = fd;
            client_count++;
            break;
        }
    }
    pthread_mutex_unlock(&clients_mutex);
}

/* 移除客户端 */
void remove_client(int fd) {
    pthread_mutex_lock(&clients_mutex);
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (client_fds[i] == fd) {
            client_fds[i] = -1;
            client_count--;
            break;
        }
    }
    pthread_mutex_unlock(&clients_mutex);
}

void* handle_client(void *arg) {
    int client_fd = *(int*)arg;
    free(arg);

    char buffer[BUFFER_SIZE];
    char msg[BUFFER_SIZE + 64];

    /* 发送欢迎消息 */
    snprintf(msg, sizeof(msg), "欢迎来到聊天室！当前在线 %d 人\n",
             client_count);
    send(client_fd, msg, strlen(msg), 0);

    /* 广播加入消息 */
    snprintf(msg, sizeof(msg), "[系统] 有新用户加入，当前在线 %d 人\n",
             client_count);
    broadcast(msg, client_fd);

    while (1) {
        memset(buffer, 0, BUFFER_SIZE);
        ssize_t bytes = recv(client_fd, buffer, BUFFER_SIZE - 1, 0);

        if (bytes <= 0) {
            break;
        }

        buffer[bytes] = '\0';
        printf("收到: %s", buffer);

        /* 广播消息 */
        snprintf(msg, sizeof(msg), "[用户%d] %s", client_fd, buffer);
        broadcast(msg, client_fd);
    }

    /* 客户端断开 */
    remove_client(client_fd);
    snprintf(msg, sizeof(msg), "[系统] 有用户离开，当前在线 %d 人\n",
             client_count);
    broadcast(msg, -1);
    printf("客户端 %d 断开，当前在线 %d 人\n", client_fd, client_count);

    close(client_fd);
    return NULL;
}

int main(void) {
    int server_fd;
    struct sockaddr_in server_addr;

    /* 初始化客户端列表 */
    for (int i = 0; i < MAX_CLIENTS; i++) {
        client_fds[i] = -1;
    }

    server_fd = socket(AF_INET, SOCK_STREAM, 0);
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = INADDR_ANY;
    server_addr.sin_port = htons(PORT);

    bind(server_fd, (struct sockaddr*)&server_addr, sizeof(server_addr));
    listen(server_fd, 10);

    printf("聊天服务器启动，端口 %d\n", PORT);

    while (1) {
        struct sockaddr_in client_addr;
        socklen_t addr_len = sizeof(client_addr);
        int client_fd = accept(server_fd,
                               (struct sockaddr*)&client_addr,
                               &addr_len);
        if (client_fd < 0) {
            perror("accept 失败");
            continue;
        }

        char ip[INET_ADDRSTRLEN];
        inet_ntop(AF_INET, &client_addr.sin_addr, ip, sizeof(ip));
        printf("新客户端: %s:%d\n", ip, ntohs(client_addr.sin_port));

        add_client(client_fd);

        int *fd_ptr = (int*)malloc(sizeof(int));
        *fd_ptr = client_fd;
        pthread_t thread;
        pthread_create(&thread, NULL, handle_client, fd_ptr);
        pthread_detach(thread);
    }

    close(server_fd);
    return 0;
}
```

## 我做得对不对？

**多进程服务器**：
1. 同时启动多个 telnet 客户端，每个都能独立通信。
2. `ps aux | grep mp_server` 可以看到多个子进程。
3. 客户端断开后，子进程正常退出。

**多线程服务器**：
1. 多个客户端同时连接，各自独立通信。
2. `top -H -p <pid>` 可以看到多个线程。

**select 服务器**：
1. 多个客户端同时连接，单线程处理全部。
2. 客户端断开后，对应的 fd 从监控列表中移除。

**epoll 服务器**：
1. 功能与 select 相同，但性能更好（仅 Linux）。
2. 大量连接时 CPU 占用远低于 select。

**聊天服务器**：
1. 多人同时连接，一人发消息其他人能看到。
2. 有人加入/离开时，所有人收到系统通知。

## 不对怎么办？

### 常见Bug 1：多进程忘记回收僵尸进程

```c
/* ❌ 子进程结束后变成僵尸，累积多了无法创建新进程 */
while (1) {
    pid_t pid = fork();
    if (pid == 0) {
        handle_client(...);
        exit(0);
    }
    /* 父进程没有 wait */
}
```

✅ 使用 `signal(SIGCHLD, handler)` 或 `waitpid(-1, NULL, WNOHANG)`。

### 常见Bug 2：select 忘记重建 fd_set

```c
/* ❌ select 会修改 fd_set，不重建会导致行为异常 */
FD_ZERO(&read_fds);
FD_SET(server_fd, &read_fds);
select(max_fd + 1, &read_fds, NULL, NULL, NULL);
/* 下次循环忘记 FD_ZERO + FD_SET */
```

✅ 每次 `select()` 调用前重建 `fd_set`。

### 常见Bug 3：epoll 的 data.fd 和实际 fd 不一致

```c
/* ❌ 关闭了 fd 但忘记从 epoll 中删除 */
close(client_fd);
/* 忘记 epoll_ctl(epfd, EPOLL_CTL_DEL, client_fd, NULL) */
```

✅ 关闭 fd 前先从 epoll 中删除 `epoll_ctl(epfd, EPOLL_CTL_DEL, fd, NULL)`。

### 常见Bug 4：广播时忘记排除发送者

```c
/* ❌ 广播消息时也发给了发送者自己 */
for (int i = 0; i < MAX; i++) {
    send(client_fds[i], msg, len, 0);  /* 发送者收到了自己的消息 */
}
```

✅ 广播时跳过 `sender_fd`。

### 常见Bug 5：聊天服务器中 client_count 的竞态条件

```c
/* ❌ 没有加锁保护 client_count 的读写 */
client_count++;  /* 非原子操作，多线程同时修改 */
```

✅ 所有对共享数据的访问都要加锁保护。

## 本章小结

| 要点 | 说明 |
|------|------|
| 单线程服务器问题 | accept/recv 阻塞，一次只能服务一个客户端 |
| 多进程并发 | fork 子进程处理每个客户端，进程隔离好但开销大 |
| 多线程并发 | 创建线程处理每个客户端，比进程轻量但共享内存 |
| IO多路复用 | 一个线程监控多个 fd，哪个就绪处理哪个 |
| select | 跨平台，最多 1024 个 fd，O(n) 复杂度 |
| poll | 类似 select，无 fd 数量限制，但仍 O(n) |
| epoll | Linux 特有，O(1) 获取就绪事件，适合高并发 |
| epoll_create1 | 创建 epoll 实例 |
| epoll_ctl | 添加/修改/删除监控的 fd |
| epoll_wait | 等待事件，返回就绪的 fd 列表 |
| 聊天服务器 | 多线程 + 广播，展示并发网络编程的完整应用 |
