## 我在做什么？

在此之前，我们写的每个 C 程序都从 `int main(void)` 开始——括号里是 `void`，表示程序不接受任何外部输入。但真正的命令行工具（如 `gcc`、`ls`、`cp`）都能接受参数：

```
gcc -Wall -o hello hello.c
ls -la /home/user
cp source.txt dest.txt
```

这些参数是怎么传到程序里的？答案藏在 `main` 函数的完整签名中：

```c
int main(int argc, char *argv[])
```

`argc` 是参数个数（argument count），`argv` 是参数值数组（argument vector）。

想象你走进一家餐厅，服务员递给你一张点菜单。`argc` 告诉你菜单上有几行，`argv[0]` 是餐厅名字，`argv[1]` 是第一道菜，`argv[2]` 是第二道菜……你的程序就是根据这张菜单决定做什么的厨师。

## 51.1 main 的完整签名

```c
/*
 * main 的两种标准写法：
 *   1. int main(void)           —— 不接受命令行参数
 *   2. int main(int argc, char *argv[])  —— 接受命令行参数
 *
 * 还可以有第三种（获取环境变量），但不常用：
 *   3. int main(int argc, char *argv[], char *envp[])
 */
```

```c
#include <stdio.h>

int main(int argc, char *argv[]) {
    printf("参数个数 argc = %d\n", argc);

    for (int i = 0; i < argc; i++) {
        printf("argv[%d] = \"%s\"\n", i, argv[i]);
    }

    return 0;
}
```

编译并运行：
```
$ gcc -o args args.c
$ ./args hello world 123

参数个数 argc = 4
argv[0] = "./args"
argv[1] = "hello"
argv[2] = "world"
argv[3] = "123"
```

关键理解：
- `argv[0]` 始终是程序名（可执行文件路径）。
- `argc` 至少为 1（因为 `argv[0]` 始终存在）。
- 所有参数都是**字符串**，即使你输入的是数字。

## 51.2 把字符串参数转为数字

所有命令行参数都是字符串。如果你需要整数或浮点数，必须手动转换。

### atoi：字符串转整数

```c
#include <stdio.h>
#include <stdlib.h>  /* atoi() */

int main(int argc, char *argv[]) {
    if (argc < 3) {
        printf("用法: %s <数字1> <数字2>\n", argv[0]);
        return 1;
    }

    int a = atoi(argv[1]);
    int b = atoi(argv[2]);

    printf("%d + %d = %d\n", a, b, a + b);

    return 0;
}
```

运行：
```
$ ./calc 10 20
10 + 20 = 30
```

**atoi 的陷阱**：如果字符串不是合法数字，`atoi("hello")` 返回 0，而且不会报错。你无法区分"用户输入了 0"和"用户输入了非法字符串"。

### atof：字符串转浮点数

```c
#include <stdio.h>
#include <stdlib.h>  /* atof() */

int main(int argc, char *argv[]) {
    if (argc < 3) {
        printf("用法: %s <数字1> <数字2>\n", argv[0]);
        return 1;
    }

    double a = atof(argv[1]);
    double b = atof(argv[2]);

    printf("%.2f + %.2f = %.2f\n", a, b, a + b);

    return 0;
}
```

### strtol：更安全的字符串转整数（带错误检测）

`strtol()` 可以检测转换是否成功，并且支持不同进制：

```c
#include <stdio.h>
#include <stdlib.h>
#include <errno.h>

int main(int argc, char *argv[]) {
    if (argc < 2) {
        printf("用法: %s <数字>\n", argv[0]);
        return 1;
    }

    char *endptr;
    errno = 0;  /* 使用前清零 */

    long value = strtol(argv[1], &endptr, 10);

    /* 检查是否发生错误 */
    if (errno == ERANGE) {
        printf("错误: 数字超出范围\n");
        return 1;
    }

    /* 检查是否真的转换了数字 */
    if (endptr == argv[1]) {
        printf("错误: \"%s\" 不是合法数字\n", argv[1]);
        return 1;
    }

    /* 检查是否有多余字符 */
    if (*endptr != '\0') {
        printf("警告: 数字后面有额外字符 \"%s\"\n", endptr);
    }

    printf("转换结果: %ld\n", value);
    return 0;
}
```

运行：
```
$ ./strtol_demo 123abc
警告: 数字后面有额外字符 "abc"
转换结果: 123

$ ./strtol_demo abc
错误: "abc" 不是合法数字

$ ./strtol_demo 99999999999999999999
错误: 数字超出范围
```

## 51.3 getopt：解析带选项的命令行参数

真正的命令行工具通常有**选项**（以 `-` 或 `--` 开头），比如 `gcc -Wall -std=c99 -o hello hello.c`。自己解析这些选项会很繁琐，POSIX 提供了 `getopt()` 函数（定义在 `<unistd.h>` 中）。

```
命令行结构:
  ./program -a -b value -c arg1 arg2
  |         |  |        |  |_________|
  |         |  |        |     非选项参数
  |         |  |        |
  |         |  |       带参数的选项 (-b value)
  |         |  |
  |         | 无参数选项 (-a)
  |_________|
     程序名
```

### getopt 基本用法

```c
/*
 * 需要包含 <unistd.h>（Linux/macOS）
 * Windows 用户可以用 MinGW 或 WSL，或者自己实现简单的解析
 * 编译: gcc -std=c99 -Wall -o getopt_demo getopt_demo.c
 */
#include <stdio.h>
#include <unistd.h>  /* getopt() */

int main(int argc, char *argv[]) {
    int opt;

    /*
     * getopt(argc, argv, "选项字符串")
     * 选项字符串中：
     *   单个字母 = 无参数选项（如 -h）
     *   字母后跟冒号 = 需要参数（如 -o filename）
     *   字母后跟双冒号 = 可选参数（极少用）
     */
    while ((opt = getopt(argc, argv, "ho:v")) != -1) {
        switch (opt) {
            case 'h':
                printf("用法: %s [-h] [-v] [-o output]\n", argv[0]);
                return 0;

            case 'o':
                printf("输出文件: %s\n", optarg);
                break;

            case 'v':
                printf("详细模式已开启\n");
                break;

            case '?':
                /* getopt 自动打印错误信息 */
                printf("未知选项或缺少参数\n");
                return 1;
        }
    }

    /* optind 指向第一个非选项参数的索引 */
    printf("\n非选项参数:\n");
    for (int i = optind; i < argc; i++) {
        printf("  argv[%d] = %s\n", i, argv[i]);
    }

    return 0;
}
```

运行：
```
$ ./getopt_demo -v -o result.txt hello world
详细模式已开启
输出文件: result.txt

非选项参数:
  argv[4] = hello
  argv[5] = world

$ ./getopt_demo -h
用法: ./getopt_demo [-h] [-v] [-o output]

$ ./getopt_demo -x
./getopt_demo: invalid option -- 'x'
未知选项或缺少参数
```

### 关键全局变量

| 变量 | 说明 |
|------|------|
| `optarg` | 指向当前选项的参数（如 `-o file` 中的 `"file"`） |
| `optind` | 下一个要处理的 `argv` 索引，初始值为 1 |
| `opterr` | 设为 0 可禁止 `getopt` 自动打印错误信息 |
| `optopt` | 当遇到未知选项时，存储该选项字符 |

## 51.4 完整示例：简易计算器

下面是一个完整的命令行计算器，支持加减乘除运算：

```c
/*
 * calculator.c —— 命令行计算器
 * 编译: gcc -std=c99 -Wall -o calculator calculator.c
 * 用法:
 *   ./calculator 10 + 20
 *   ./calculator 100 - 50
 *   ./calculator 7 x 8
 *   ./calculator 100 / 3
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(int argc, char *argv[]) {
    if (argc != 4) {
        printf("用法: %s <数字1> <运算符> <数字2>\n", argv[0]);
        printf("支持的运算符: + - x /\n");
        printf("示例: %s 10 + 20\n", argv[0]);
        return 1;
    }

    /* 解析数字 */
    double a = atof(argv[1]);
    double b = atof(argv[3]);
    const char *op = argv[2];

    double result;

    if (strcmp(op, "+") == 0) {
        result = a + b;
    } else if (strcmp(op, "-") == 0) {
        result = a - b;
    } else if (strcmp(op, "x") == 0) {
        result = a * b;
    } else if (strcmp(op, "/") == 0) {
        if (b == 0.0) {
            printf("错误: 除数不能为零\n");
            return 1;
        }
        result = a / b;
    } else {
        printf("错误: 不支持的运算符 \"%s\"\n", op);
        printf("支持的运算符: + - x /\n");
        return 1;
    }

    printf("%.2f %s %.2f = %.2f\n", a, op, b, result);
    return 0;
}
```

## 51.5 完整代码清单

```c
/*
 * cli_args_demo.c —— 命令行参数完整演示
 * 编译: gcc -std=c99 -Wall -o cli_demo cli_args_demo.c
 *
 * 功能:
 *   - 打印所有命令行参数
 *   - 安全转换字符串为整数（strtol）
 *   - 使用 getopt 解析选项
 *   - 简易计算器
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <errno.h>

#ifdef __unix__
#include <unistd.h>  /* getopt() in POSIX */
#endif

/* 打印所有命令行参数 */
void print_args(int argc, char *argv[]) {
    printf("=== 命令行参数列表 ===\n");
    printf("argc = %d\n", argc);
    for (int i = 0; i < argc; i++) {
        printf("  argv[%d] = \"%s\"\n", i, argv[i]);
    }
    printf("\n");
}

/* 安全转换字符串为 long */
int safe_strtol(const char *str, long *out) {
    char *endptr;
    errno = 0;
    *out = strtol(str, &endptr, 10);

    if (errno == ERANGE) {
        fprintf(stderr, "错误: 数字 \"%s\" 超出范围\n", str);
        return -1;
    }
    if (endptr == str) {
        fprintf(stderr, "错误: \"%s\" 不是合法数字\n", str);
        return -1;
    }
    if (*endptr != '\0') {
        fprintf(stderr, "警告: \"%s\" 后面有无效字符 \"%s\"\n", str, endptr);
    }
    return 0;
}

/* 简易计算器功能 */
int calculator_mode(int argc, char *argv[]) {
    if (argc != 4) {
        printf("计算器模式: %s <数字1> <运算符> <数字2>\n", argv[0]);
        printf("运算符: + - x /\n");
        return 1;
    }

    double a = atof(argv[1]);
    double b = atof(argv[3]);
    const char *op = argv[2];
    double result;

    if (strcmp(op, "+") == 0) {
        result = a + b;
    } else if (strcmp(op, "-") == 0) {
        result = a - b;
    } else if (strcmp(op, "x") == 0) {
        result = a * b;
    } else if (strcmp(op, "/") == 0) {
        if (b == 0.0) {
            printf("错误: 除数不能为零\n");
            return 1;
        }
        result = a / b;
    } else {
        printf("错误: 不支持的运算符 \"%s\"\n", op);
        return 1;
    }

    printf("%.2f %s %.2f = %.2f\n", a, op, b, result);
    return 0;
}

/* 打印帮助信息 */
void print_help(const char *progname) {
    printf("用法: %s [选项] [参数...]\n\n", progname);
    printf("选项:\n");
    printf("  -h          显示帮助信息\n");
    printf("  -c          计算器模式（需要三个参数）\n");
    printf("  -s <数字>   安全转换字符串为整数\n");
    printf("  -a          打印所有参数\n");
    printf("\n");
    printf("示例:\n");
    printf("  %s -a hello world\n", progname);
    printf("  %s -c 10 + 20\n", progname);
    printf("  %s -s 123abc\n", progname);
}

int main(int argc, char *argv[]) {
    if (argc < 2) {
        print_help(argv[0]);
        return 0;
    }

    int opt;
    int show_all = 0;

#ifdef __unix__
    while ((opt = getopt(argc, argv, "hcas:")) != -1) {
        switch (opt) {
            case 'h':
                print_help(argv[0]);
                return 0;

            case 'a':
                show_all = 1;
                break;

            case 'c':
                /* 计算器模式：剩余参数传给 calculator_mode */
                return calculator_mode(argc - optind + 1, argv + optind - 1);

            case 's': {
                long value;
                if (safe_strtol(optarg, &value) == 0) {
                    printf("转换结果: %ld\n", value);
                }
                break;
            }

            case '?':
                return 1;
        }
    }
#else
    /* Windows 上没有 getopt，使用简单的手动解析 */
    for (int i = 1; i < argc; i++) {
        if (strcmp(argv[i], "-h") == 0) {
            print_help(argv[0]);
            return 0;
        } else if (strcmp(argv[i], "-a") == 0) {
            show_all = 1;
        } else if (strcmp(argv[i], "-c") == 0) {
            if (i + 3 < argc) {
                return calculator_mode(4, &argv[i]);
            } else {
                printf("计算器模式需要三个参数\n");
                return 1;
            }
        } else if (strcmp(argv[i], "-s") == 0) {
            if (i + 1 < argc) {
                long value;
                if (safe_strtol(argv[i + 1], &value) == 0) {
                    printf("转换结果: %ld\n", value);
                }
                i++;
            } else {
                printf("-s 需要一个参数\n");
                return 1;
            }
        }
    }
#endif

    if (show_all) {
        print_args(argc, argv);
    }

    return 0;
}
```

## 我做得对不对？

1. 不带参数运行 `./cli_demo`，显示帮助信息。
2. `./cli_demo -a hello world` 打印所有参数，`argc` 正确，`argv[0]` 是程序名。
3. `./cli_demo -c 10 + 20` 输出 "10.00 + 20.00 = 30.00"。
4. `./cli_demo -c 100 / 0` 输出 "错误: 除数不能为零"。
5. `./cli_demo -s 123abc` 输出警告但转换成功；`./cli_demo -s abc` 报错。

---

## 不对怎么办？

### 常见Bug 1：argv 索引越界

```c
/* ❌ argc 可能是 1（只有程序名），访问 argv[1] 越界 */
int main(int argc, char *argv[]) {
    printf("%s\n", argv[1]);  /* 段错误！ */
}
```

✅ 访问前检查 `argc > 1`。

### 常见Bug 2：atoi 无法区分 0 和非法输入

```c
/* ❌ 用户输入 "abc" 时 atoi 返回 0，程序静默错误 */
int value = atoi(argv[1]);
printf("两倍: %d\n", value * 2);
```

✅ 使用 `strtol()` 并检查 `endptr`。

### 常见Bug 3：getopt 选项字符串写错

```c
/* ❌ 忘记冒号，-o 应该是带参数的选项 */
while ((opt = getopt(argc, argv, "ho")) != -1) {  /* 'o' 后面没有冒号 */
    case 'o':
        printf("%s\n", optarg);  /* optarg 是垃圾值 */
}
```

✅ 带参数的选项后面加冒号：`"ho:"`。

### 常见Bug 4：忘记处理 getopt 返回的 '?'

```c
/* ❌ 没有 case '?'，用户输入错误选项时行为未定义 */
switch (opt) {
    case 'h': /* ... */
    case 'v': /* ... */
    /* 缺少 default 或 case '?' */
}
```

✅ 加上 `case '?': return 1;` 或 `default: /* 处理 */`。

### 常见Bug 5：optind 未重置导致多次解析出问题

```c
/* ❌ 如果要多次调用 getopt，需要重置 optind */
optind = 1;  /* 重置到起始位置 */
```

✅ 在第二次调用 `getopt()` 前将 `optind` 重置为 1。

## 本章小结

| 要点 | 说明 |
|------|------|
| main 完整签名 | `int main(int argc, char *argv[])` |
| argc | 命令行参数个数，至少为 1 |
| argv[0] | 始终是程序名/路径 |
| argv[i] | 第 i 个参数，全部是字符串 |
| atoi() | 字符串转整数，无法区分 0 和非法输入 |
| atof() | 字符串转浮点数 |
| strtol() | 安全转换，支持错误检测和进制指定 |
| getopt() | 解析 `-x` 和 `-x value` 格式的选项 |
| optarg | 当前选项的参数值 |
| optind | 下一个非选项参数在 argv 中的索引 |
