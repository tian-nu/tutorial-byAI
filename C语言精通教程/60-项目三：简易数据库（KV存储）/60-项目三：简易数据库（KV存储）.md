## 我在做什么？

数据库听起来很复杂，但它的核心就是"存"和"取"。本章你将实现一个最简单的键值数据库——像一个超级字典，用 `SET` 存入键值对，用 `GET` 取出，用 `DEL` 删除。数据保存在文件中，程序关闭后不会丢失。


## 60.3 数据结构设计

### 哈希表

*哈希表\*此术语见附录F\** 是键值存储的核心数据结构。它通过哈希函数将键映射到数组索引，实现 O(1) 的平均查找时间。

```
索引     桶（链表）
[0]  →  NULL
[1]  →  ("name","Alice") → NULL
[2]  →  ("age","25") → ("city","Beijing") → NULL
[3]  →  NULL
...
```

相同的哈希值用链表串联（链地址法）。

### 结构体定义

```c
// hash.h
#ifndef HASH_H
#define HASH_H

#define TABLE_SIZE 256           // 哈希表桶数量
#define MAX_KEY_LEN   128
#define MAX_VALUE_LEN 512

// 键值对节点
typedef struct Entry {
    char key[MAX_KEY_LEN];
    char value[MAX_VALUE_LEN];
    struct Entry *next;          // 链表下一个节点
} Entry;

// 哈希表
typedef struct {
    Entry *buckets[TABLE_SIZE];  // 桶数组
    int count;                   // 当前键值对数量
} HashTable;

// 哈希函数
unsigned int hash_function(const char *key);

// 哈希表操作
void ht_init(HashTable *ht);
int  ht_set(HashTable *ht, const char *key, const char *value);
const char* ht_get(HashTable *ht, const char *key);
int  ht_del(HashTable *ht, const char *key);
void ht_keys(HashTable *ht, char keys[][MAX_KEY_LEN], int *count);
void ht_free(HashTable *ht);

#endif
```

## 60.4 实现步骤

### 步骤 1：哈希函数

```c
// hash.c - 哈希函数
#include <string.h>
#include <stdlib.h>
#include "hash.h"

// djb2 哈希算法（由 Daniel J. Bernstein 发明，简单高效）
unsigned int hash_function(const char *key) {
    unsigned int hash = 5381;
    int c;
    while ((c = *key++)) {
        hash = ((hash << 5) + hash) + c;  // hash * 33 + c
    }
    return hash % TABLE_SIZE;
}
```

### 步骤 2：哈希表基本操作

```c
// hash.c - 哈希表操作（续）

void ht_init(HashTable *ht) {
    for (int i = 0; i < TABLE_SIZE; i++) {
        ht->buckets[i] = NULL;
    }
    ht->count = 0;
}

// SET 操作：插入或更新键值对
int ht_set(HashTable *ht, const char *key, const char *value) {
    unsigned int idx = hash_function(key);
    Entry *cur = ht->buckets[idx];

    // 查找是否已存在同名 key
    while (cur) {
        if (strcmp(cur->key, key) == 0) {
            // 更新值
            strncpy(cur->value, value, MAX_VALUE_LEN - 1);
            cur->value[MAX_VALUE_LEN - 1] = '\0';
            return 0;  // 更新成功
        }
        cur = cur->next;
    }

    // 不存在，创建新节点插入链表头部
    Entry *new_entry = malloc(sizeof(Entry));
    if (!new_entry) return -1;

    strncpy(new_entry->key, key, MAX_KEY_LEN - 1);
    new_entry->key[MAX_KEY_LEN - 1] = '\0';
    strncpy(new_entry->value, value, MAX_VALUE_LEN - 1);
    new_entry->value[MAX_VALUE_LEN - 1] = '\0';

    // 头插法：新节点放在链表最前面
    new_entry->next = ht->buckets[idx];
    ht->buckets[idx] = new_entry;
    ht->count++;

    return 1;  // 新增成功
}

// GET 操作：查找键对应的值
const char* ht_get(HashTable *ht, const char *key) {
    unsigned int idx = hash_function(key);
    Entry *cur = ht->buckets[idx];

    while (cur) {
        if (strcmp(cur->key, key) == 0) {
            return cur->value;
        }
        cur = cur->next;
    }
    return NULL;  // 未找到
}

// DEL 操作：删除键值对
int ht_del(HashTable *ht, const char *key) {
    unsigned int idx = hash_function(key);
    Entry *cur = ht->buckets[idx];
    Entry *prev = NULL;

    while (cur) {
        if (strcmp(cur->key, key) == 0) {
            if (prev == NULL) {
                ht->buckets[idx] = cur->next;
            } else {
                prev->next = cur->next;
            }
            free(cur);
            ht->count--;
            return 1;
        }
        prev = cur;
        cur = cur->next;
    }
    return 0;  // 未找到
}

// 列出所有键
void ht_keys(HashTable *ht, char keys[][MAX_KEY_LEN], int *out_count) {
    *out_count = 0;
    for (int i = 0; i < TABLE_SIZE; i++) {
        Entry *cur = ht->buckets[i];
        while (cur) {
            strncpy(keys[*out_count], cur->key, MAX_KEY_LEN - 1);
            keys[*out_count][MAX_KEY_LEN - 1] = '\0';
            (*out_count)++;
            cur = cur->next;
        }
    }
}

// 释放哈希表所有内存
void ht_free(HashTable *ht) {
    for (int i = 0; i < TABLE_SIZE; i++) {
        Entry *cur = ht->buckets[i];
        while (cur) {
            Entry *next = cur->next;
            free(cur);
            cur = next;
        }
        ht->buckets[i] = NULL;
    }
    ht->count = 0;
}
```

### 步骤 3：文件持久化模块（kvdb.c）

```c
// kvdb.h
#ifndef KVDB_H
#define KVDB_H

#include "hash.h"

#define DB_FILENAME "kvdb.dat"

int  kvdb_load(HashTable *ht);
int  kvdb_save(HashTable *ht);
```

```c
// kvdb.c
#include <stdio.h>
#include <string.h>
#include "kvdb.h"

// 从文件加载数据到哈希表
int kvdb_load(HashTable *ht) {
    FILE *fp = fopen(DB_FILENAME, "r");
    if (!fp) return 0;  // 文件不存在，从空数据库开始

    char line[MAX_KEY_LEN + MAX_VALUE_LEN + 4];
    while (fgets(line, sizeof(line), fp)) {
        // 去掉末尾换行
        size_t len = strlen(line);
        if (len > 0 && line[len - 1] == '\n') {
            line[len - 1] = '\0';
        }

        char key[MAX_KEY_LEN] = {0};
        char value[MAX_VALUE_LEN] = {0};

        // 解析 key|value
        char *sep = strchr(line, '|');
        if (sep) {
            size_t key_len = sep - line;
            if (key_len >= MAX_KEY_LEN) key_len = MAX_KEY_LEN - 1;
            strncpy(key, line, key_len);
            key[key_len] = '\0';

            strncpy(value, sep + 1, MAX_VALUE_LEN - 1);
            value[MAX_VALUE_LEN - 1] = '\0';

            ht_set(ht, key, value);
        }
    }

    fclose(fp);
    return 1;
}

// 保存哈希表到文件
int kvdb_save(HashTable *ht) {
    FILE *fp = fopen(DB_FILENAME, "w");
    if (!fp) {
        fprintf(stderr, "无法打开文件 %s 进行写入\n", DB_FILENAME);
        return -1;
    }

    for (int i = 0; i < TABLE_SIZE; i++) {
        Entry *cur = ht->buckets[i];
        while (cur) {
            fprintf(fp, "%s|%s\n", cur->key, cur->value);
            cur = cur->next;
        }
    }

    fclose(fp);
    printf("已保存 %d 条记录到 %s\n", ht->count, DB_FILENAME);
    return 0;
}
```

### 步骤 4：命令行交互（main.c）

```c
// main.c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include "kvdb.h"
#include "hash.h"

static void trim_newline(char *str) {
    size_t len = strlen(str);
    if (len > 0 && str[len - 1] == '\n') {
        str[len - 1] = '\0';
    }
}

static void print_help(void) {
    printf("KV 数据库 (kvdb) 命令列表:\n");
    printf("  SET <key> <value>   设置键值对\n");
    printf("  GET <key>           获取键对应的值\n");
    printf("  DEL <key>           删除键值对\n");
    printf("  KEYS                列出所有键\n");
    printf("  COUNT               显示键值对数量\n");
    printf("  SAVE                手动保存到文件\n");
    printf("  HELP                显示本帮助\n");
    printf("  QUIT                退出并保存\n");
}

int main(void) {
    HashTable ht;
    ht_init(&ht);
    kvdb_load(&ht);

    printf("简易 KV 数据库 v1.0（输入 HELP 查看帮助）\n");

    char input[1024];
    while (1) {
        printf("\nkvdb> ");
        if (!fgets(input, sizeof(input), stdin)) break;
        trim_newline(input);

        if (strlen(input) == 0) continue;

        char cmd[32] = {0};
        char key[MAX_KEY_LEN] = {0};
        char value[MAX_VALUE_LEN] = {0};

        // 解析命令
        // 格式: CMD KEY VALUE
        int n = sscanf(input, "%31s %127s %511[^\n]", cmd, key, value);

        if (strcmp(cmd, "QUIT") == 0) {
            kvdb_save(&ht);
            ht_free(&ht);
            printf("再见！\n");
            break;
        } else if (strcmp(cmd, "HELP") == 0) {
            print_help();
        } else if (strcmp(cmd, "SET") == 0) {
            if (n < 3) {
                printf("用法: SET <key> <value>\n");
                continue;
            }
            int result = ht_set(&ht, key, value);
            if (result == 1) {
                printf("OK (新增)\n");
            } else if (result == 0) {
                printf("OK (更新)\n");
            } else {
                printf("错误: 内存不足\n");
            }
        } else if (strcmp(cmd, "GET") == 0) {
            if (n < 2) {
                printf("用法: GET <key>\n");
                continue;
            }
            const char *val = ht_get(&ht, key);
            if (val) {
                printf("\"%s\"\n", val);
            } else {
                printf("(nil)\n");
            }
        } else if (strcmp(cmd, "DEL") == 0) {
            if (n < 2) {
                printf("用法: DEL <key>\n");
                continue;
            }
            if (ht_del(&ht, key)) {
                printf("OK (已删除)\n");
            } else {
                printf("(nil) — 键不存在\n");
            }
        } else if (strcmp(cmd, "KEYS") == 0) {
            char keys[TABLE_SIZE][MAX_KEY_LEN];
            int count = 0;
            ht_keys(&ht, keys, &count);
            if (count == 0) {
                printf("(空数据库)\n");
            } else {
                for (int i = 0; i < count; i++) {
                    printf("%d) %s\n", i + 1, keys[i]);
                }
            }
        } else if (strcmp(cmd, "COUNT") == 0) {
            printf("%d\n", ht.count);
        } else if (strcmp(cmd, "SAVE") == 0) {
            kvdb_save(&ht);
        } else {
            printf("未知命令: %s（输入 HELP 查看帮助）\n", cmd);
        }
    }

    return 0;
}
```

## 60.6 常见错误对照

| 错误写法 | 问题 | 正确写法 |
|----------|------|----------|
| 哈希函数返回 `hash` 不取模 | 索引超出数组范围 | `return hash % TABLE_SIZE;` |
| SET 时用尾插法 | 每次都要遍历整个链表，效率低 | 头插法：`new_entry->next = ht->buckets[idx];` |
| DEL 后忘记 `free` | 内存泄漏 | 先更新指针，再 `free(cur)` |
| 文件加载时 `fgets` 未去除 `\n` | key 末尾带换行符，GET 不到 | 加载后 `line[strlen(line)-1] = '\0'` |
| `sscanf` 解析 `SET key value` 时 value 含空格 | value 只读到第一个空格 | 使用 `%[^\n]` 读取整行剩余部分 |

## 60.8 三问

1. **为什么用哈希表而不是普通数组？** 数组查找需要遍历，O(n) 时间。哈希表通过哈希函数直接定位到桶，平均 O(1) 时间。当数据量增大时，差距会非常明显。

2. **哈希冲突怎么处理？** 本实现使用链地址法——同一个桶的多个键值对用链表串联。另一种常见方法是开放地址法（冲突时找下一个空位）。链地址法实现简单，删除方便。

3. **为什么用 `djb2` 哈希算法？** 它简单（几行代码）、快速（位运算和加法）、分布均匀（对常见字符串输入）。对于教学项目足够。生产环境可考虑 MurmurHash 或 SipHash。
