## 98.1 短链接系统

### 题目描述

设计一个类似"TinyURL"的短链接服务。用户提交一个长URL（如 `https://example.com/very/long/path/to/article`），系统返回一个短URL（如 `https://t.cn/A3x9K`）。用户访问短URL时，自动跳转到原始长URL。

**规模估算**：每天生成100万个短链接，读写比例100:1（读远大于写）。

### 短链接生成算法

核心问题：如何把长URL映射到一个短码？

**Hash + Base62方案**：

```
长URL: https://www.example.com/article/12345
  ↓
MD5("https://www.example.com/article/12345") = "d41d8cd9..."
  ↓
取前7个字节 → 转Base62编码
  ↓
短码: "A3x9Km2"
  ↓
短链接: https://t.cn/A3x9Km2
```

为什么用Base62？Base62 = `[0-9a-zA-Z]`，共62个字符。7位Base62 = 62^7 ≈ 3.5万亿种组合——足够每天100万用100年。

为什么不用Base64？Base64包含 `+` 和 `/`，在URL中需要转义，麻烦。

**Go实现**：

```go
import (
	"crypto/md5"
	"encoding/hex"
	"strings"
)

const base62Chars = "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ"

func padZero(s string, length int) string {
	if len(s) >= length {
		return s[:length]
	}
	return strings.Repeat("0", length-len(s)) + s
}

func encodeBase62(num uint64) string {
	if num == 0 {
		return "0"
	}
	var result []byte
	for num > 0 {
		result = append([]byte{base62Chars[num%62]}, result...)
		num /= 62
	}
	return string(result)
}

func generateShortCode(url string) string {
	hash := md5.Sum([]byte(url))

	var num uint64
	for i := 0; i < 7; i++ {
		num = num*256 + uint64(hash[i])
	}

	return padZero(encodeBase62(num), 7)
}
```

**创建短链接**：

```go
func CreateShortURL(w http.ResponseWriter, r *http.Request) {
	var req struct {
		LongURL string `json:"long_url"`
	}
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		http.Error(w, "请求体格式错误", 400)
		return
	}

	if req.LongURL == "" {
		http.Error(w, "URL不能为空", 400)
		return
	}

	shortCode := generateShortCode(req.LongURL + strconv.FormatInt(time.Now().UnixNano(), 10))

	err := db.Exec("INSERT INTO short_urls (short_code, long_url) VALUES (?, ?)",
		shortCode, req.LongURL).Error
	if err != nil {
		http.Error(w, "生成失败", 500)
		return
	}

	rdb.Set(ctx, "url:"+shortCode, req.LongURL, 30*24*time.Hour)

	json.NewEncoder(w).Encode(map[string]string{
		"short_url": "https://t.cn/" + shortCode,
	})
}
```

**短链接跳转**：

```go
func RedirectShortURL(w http.ResponseWriter, r *http.Request) {
	shortCode := strings.TrimPrefix(r.URL.Path, "/")

	longURL, err := rdb.Get(ctx, "url:"+shortCode).Result()
	if err != nil {
		var record ShortURL
		err = db.Where("short_code = ?", shortCode).First(&record).Error
		if err != nil {
			http.NotFound(w, r)
			return
		}
		longURL = record.LongURL
		rdb.Set(ctx, "url:"+shortCode, longURL, 30*24*time.Hour)
	}

	go db.Exec("UPDATE short_urls SET click_count = click_count + 1 WHERE short_code = ?", shortCode)

	http.Redirect(w, r, longURL, http.StatusFound)
}
```

## 98.2 秒杀系统

### 题目描述

设计一个秒杀（Flash Sale）系统。特定时间点，数十万用户同时抢购少量商品（如1000件库存卖1000万元）。要求：不能超卖、系统不崩溃、用户体验可接受。

### 秒杀全链路架构

```
用户浏览器
    │
    │ ① 前端层：验证码 + 答题 + 按钮防重复点击
    │
    ▼
Nginx (静态化 + 限量)
    │
    │ ② 网关层：规则过滤 + Token校验 + 限流（令牌桶）
    │
    ▼
秒杀API服务
    │
    │ ③ 服务层：
    │   - 校验秒杀时间窗口
    │   - 校验用户资格（是否登录、是否已抢过）
    │   - Redis预减库存（Lua脚本原子操作）
    │   - 发送MQ消息（异步下单）
    │   - 返回"排队中，稍后查看结果"
    │
    ├──→ Redis：库存预减 + 抢购资格记录
    │
    └──→ RabbitMQ：异步下单队列
              │
              ▼
         订单消费者 (消费MQ消息)
              │
              │ ④ 后端层：
              │   - 校验真实库存
              │   - 创建订单
              │   - 更新订单状态
              │   - 返回抢购结果
              │
              ▼
           MySQL (最终订单数据)
```

### 为什么这样设计

**① 前端层：消减无效流量**

- **验证码/答题**：过滤掉机器人和脚本（80%的瞬时流量是机器发起的）
- **按钮防重复**：点一次后按钮变灰，防止用户疯狂点击产生10次请求

**② 网关层：限流**

```go
limiter := rate.NewLimiter(1000, 2000)

func seckillGate(w http.ResponseWriter, r *http.Request) {
    if !limiter.Allow() {
        http.Error(w, "系统繁忙", 429)
        return
    }
    proxy.ServeHTTP(w, r)
}
```

即使每秒10万请求进来，网关只放1000个过去。剩下的直接429。

**③ Redis预减库存（核心）**

为什么用Redis而不是MySQL？

MySQL单行更新能扛5000QPS左右。秒杀场景瞬间可能有10万请求——MySQL直接被打死。

Redis单机能扛10万QPS，而且用Lua脚本可以保证"检查库存"和"扣库存"两个操作的原子性：

```lua
-- seckill.lua
local key = KEYS[1]
local stock = tonumber(redis.call('get', key))
if stock == nil or stock <= 0 then
    return 0
end
redis.call('decr', key)
return 1
```

**Go调用Lua脚本**：

```go
var seckillScript = redis.NewScript(`
    local key = KEYS[1]
    local stock = tonumber(redis.call('get', key))
    if stock == nil or stock <= 0 then
        return 0
    end
    redis.call('decr', key)
    return 1
`)

func TrySeckill(productID, userID string) (bool, error) {
    result, err := seckillScript.Run(ctx, rdb, []string{"seckill:stock:" + productID}).Result()
    if err != nil {
        return false, err
    }
    success := result.(int64) == 1
    if !success {
        return false, nil
    }

    hasGrabbed, _ := rdb.SetNX(ctx, "seckill:user:"+productID+":"+userID, "1", 10*time.Minute).Result()
    if !hasGrabbed {
        rdb.Incr(ctx, "seckill:stock:"+productID)
        return false, nil
    }

    mq.Publish("seckill_orders", SeckillOrder{
        ProductID: productID,
        UserID:    userID,
    })

    return true, nil
}
```

**④ 异步下单**

Redis扣库存成功后，不直接创建订单（避免等数据库），而是发一条MQ消息然后告诉用户"排队中"。消费者在后端慢慢处理：

```go
func StartSeckillOrderConsumer() {
    msgs, _ := ch.Consume("seckill_orders", "", false, false, false, false, nil)

    go func() {
        for d := range msgs {
            var order SeckillOrder
            json.Unmarshal(d.Body, &order)

            tx := db.Begin()
            err := tx.Exec("INSERT INTO orders (...) VALUES (...)", order).Error
            if err != nil {
                tx.Rollback()
                rdb.Incr(ctx, "seckill:stock:"+order.ProductID)
                d.Ack(false)
                continue
            }
            tx.Commit()
            d.Ack(false)
        }
    }()
}
```

## 98.3 消息推送系统

### 题目描述

### 长连接管理

使用WebSocket。服务端维护"用户ID→连接"的映射。

**Go WebSocket连接管理**：

```go
import "github.com/gorilla/websocket"

type ConnectionManager struct {
	connections map[string]*websocket.Conn
	mu          sync.RWMutex
}

var manager = &ConnectionManager{
	connections: make(map[string]*websocket.Conn),
}

func (m *ConnectionManager) Register(userID string, conn *websocket.Conn) {
	m.mu.Lock()
	defer m.mu.Unlock()

	if oldConn, ok := m.connections[userID]; ok {
		oldConn.Close()
	}
	m.connections[userID] = conn
}

func (m *ConnectionManager) Remove(userID string) {
	m.mu.Lock()
	defer m.mu.Unlock()
	delete(m.connections, userID)
}

func (m *ConnectionManager) SendToUser(userID string, message []byte) error {
	m.mu.RLock()
	defer m.mu.RUnlock()

	conn, ok := m.connections[userID]
	if !ok {
		return fmt.Errorf("用户不在线")
	}
	return conn.WriteMessage(websocket.TextMessage, message)
}
```

### 心跳机制

客户端每隔30秒发一次心跳（Ping），服务器5秒内没收到就认为连接断开。

```go
func handleConn(userID string, conn *websocket.Conn) {
	manager.Register(userID, conn)
	defer manager.Remove(userID)

	conn.SetReadDeadline(time.Now().Add(60 * time.Second))
	conn.SetPongHandler(func(string) error {
		conn.SetReadDeadline(time.Now().Add(60 * time.Second))
		return nil
	})

	go func() {
		ticker := time.NewTicker(30 * time.Second)
		defer ticker.Stop()
		for range ticker.C {
			if err := conn.WriteMessage(websocket.PingMessage, nil); err != nil {
				return
			}
		}
	}()

	for {
		_, message, err := conn.ReadMessage()
		if err != nil {
			break
		}
		processMessage(userID, message)
	}
}
```

### 分片（解决单机连接数上限）

一台服务器最多维持几万个WebSocket连接。千万用户需要几百台机器。按用户ID哈希分片：

```go
func getServerNode(userID string) string {
    hash := fnv.New32a()
    hash.Write([]byte(userID))
    index := hash.Sum32() % uint32(len(serverNodes))
    return serverNodes[index]
}
```

发送消息时，先确定目标用户在哪个分片，然后把消息路由到那个分片。

### 离线消息

查询用户在线状态：
- 在线 → 直接推送
- 离线 → 消息存入离线表，用户下次上线时拉取

```sql
CREATE TABLE offline_messages (
    id         BIGINT AUTO_INCREMENT PRIMARY KEY,

```

用户上线后拉取未读消息并标记已读。

## 98.4 Feed流（信息流）

### 题目描述

设计一个类似微博/朋友圈的Feed流系统。用户关注了其他人，在自己的首页看到所有关注者的最新动态（按时间倒序）。

关注者的新动态写入后，所有人都能在自己的Feed中看到。

### 三种方案

**推模式（Push / Fan-out on Write）**：

当用户A发布了一条动态→立刻推送到所有关注A的用户的Feed列表中。

```
A发动态
  │
  ├─→ 写入 关注者B的收件箱
  ├─→ 写入 关注者C的收件箱
  ├─→ 写入 关注者D的收件箱
  └─→ ...（如果A有100万粉丝，就要写100万次）
```

优点：读Feed时极快（直接从自己的收件箱拿数据）
缺点：写放大严重（一个明星发一条动态，100万粉丝就是100万次写入）

**拉模式（Pull / Fan-out on Read）**：

用户打开Feed→实时去所有关注者的发件箱中拉取，合并排序。

优点：没有写放大，存储简单
缺点：读慢（要查N个关注者的数据后合并）

**推拉结合（Push+Pull，业界最常用）**：

```
用户分类：
  大V用户（粉丝>10万）：拉模式
     发动态时 → 存到自己的发件箱 → 粉丝来读时去"拉"
  普通用户（粉丝<10万）：推模式
     发动态时 → 推送到所有粉丝的收件箱

用户打开Feed：
  从自己的"收件箱"获取（被推送进来的普通用户动态）
  + 去自己关注的每个大V的"发件箱"拉取
  + 合并排序 → 返回
```

### 核心表设计（Redis实现）

```
普通粉丝 → 收件箱：
  feed:inbox:{userID} → Sorted Set (score=时间戳, member=动态ID)

大V → 发件箱：
  feed:outbox:{vipID} → Sorted Set (score=时间戳, member=动态ID)

大V列表（每个人关注的大V）：
  feed:vips:{userID} → Set
```

**Go关键代码**：

```go
func PublishFeed(userID string, feed Feed) error {
	feedID := generateFeedID()
	feedJSON, _ := json.Marshal(feed)

	feedDataKey := "feed:data:" + feedID
	rdb.Set(ctx, feedDataKey, feedJSON, 30*24*time.Hour)

	isVIP, _ := rdb.Get(ctx, "feed:is_vip:"+userID).Result()
	if isVIP == "1" {
		rdb.ZAdd(ctx, "feed:outbox:"+userID, redis.Z{
			Score:  float64(feed.Timestamp),
			Member: feedID,
		})
		return nil
	}

	followerIDs, _ := rdb.SMembers(ctx, "followers:"+userID).Result()
	pipe := rdb.Pipeline()
	for _, followerID := range followerIDs {
		pipe.ZAdd(ctx, "feed:inbox:"+followerID, redis.Z{
			Score:  float64(feed.Timestamp),
			Member: feedID,
		})
		pipe.ZRemRangeByRank(ctx, "feed:inbox:"+followerID, 0, -501)
	}
	_, err := pipe.Exec(ctx)
	return err
}

func ReadFeed(userID string, cursor int64, count int) ([]Feed, error) {
	inboxKey := "feed:inbox:" + userID
	vipIDs, _ := rdb.SMembers(ctx, "feed:vips:"+userID).Result()

	pipe := rdb.Pipeline()
	pipe.ZRevRangeByScore(inboxKey, &redis.ZRangeBy{
		Min: "0", Max: "+inf", Offset: cursor, Count: int64(count * 2),
	})

	for _, vipID := range vipIDs {
		pipe.ZRevRangeByScore("feed:outbox:"+vipID, &redis.ZRangeBy{
			Min: "0", Max: "+inf", Offset: 0, Count: 20,
		})
	}

	cmders, _ := pipe.Exec(ctx)

	var feedIDs []string
	for _, cmd := range cmders {
		ids, _ := cmd.(*redis.StringSliceCmd).Result()
		feedIDs = append(feedIDs, ids...)
	}

	dedup := make(map[string]bool)
	var feeds []Feed
	for _, feedID := range feedIDs {
		if dedup[feedID] {
			continue
		}
		dedup[feedID] = true

		data, _ := rdb.Get(ctx, "feed:data:"+feedID).Result()
		var feed Feed
		json.Unmarshal([]byte(data), &feed)
		feeds = append(feeds, feed)
	}

	return feeds, nil
}
```
