```go
package rabbitmq

import (
	"log"
	"sync"

	amqp "github.com/rabbitmq/amqp091-go"
)

var (
	conn *amqp.Connection
	once sync.Once
)

func GetConnection() *amqp.Connection {
	once.Do(func() {
		var err error
		conn, err = amqp.Dial("amqp://admin:admin123@localhost:5672/")
		if err != nil {
			log.Fatalf("RabbitMQ连接失败: %v", err)
		}
		log.Println("RabbitMQ已连接")
	})
	return conn
}

func GetChannel() *amqp.Channel {
	ch, err := GetConnection().Channel()
	if err != nil {
		log.Fatalf("打开Channel失败: %v", err)
	}
	return ch
}
```

### 交换机与队列声明

在整个系统中，我们定义以下交换机：

```
order_exchange (topic)
  ├── order.created  →  routing_key="order.created"
  └── order.paid     →  routing_key="order.paid"

notification_exchange (fanout)
  └── 广播给所有通知消费者
```

```go
func SetupRabbitMQ(ch *amqp.Channel) error {
	err := ch.ExchangeDeclare("order_exchange", "topic", true, false, false, false, nil)
	if err != nil {
		return err
	}

	err = ch.ExchangeDeclare("notification_exchange", "fanout", true, false, false, false, nil)
	if err != nil {
		return err
	}

	inventoryQueue, err := ch.QueueDeclare("inventory_queue", true, false, false, false, nil)
	if err != nil {
		return err
	}
	err = ch.QueueBind(inventoryQueue.Name, "order.created", "order_exchange", false, nil)
	if err != nil {
		return err
	}

	smsQueue, err := ch.QueueDeclare("sms_queue", true, false, false, false, nil)
	if err != nil {
		return err
	}
	err = ch.QueueBind(smsQueue.Name, "", "notification_exchange", false, nil)
	if err != nil {
		return err
	}

	emailQueue, err := ch.QueueDeclare("email_queue", true, false, false, false, nil)
	if err != nil {
		return err
	}
	err = ch.QueueBind(emailQueue.Name, "", "notification_exchange", false, nil)
	if err != nil {
		return err
	}

	return nil
}
```

### 生产者：下单服务

```go
package producer

import (
	"context"
	"encoding/json"
	"log"
	"time"

	amqp "github.com/rabbitmq/amqp091-go"
	"async-order/model"
	"async-order/rabbitmq"
)

func PublishOrderCreated(order model.Order) error {
	ch := rabbitmq.GetChannel()
	defer ch.Close()

	body, err := json.Marshal(order)
	if err != nil {
		return err
	}

	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()

	err = ch.PublishWithContext(
		ctx,
		"order_exchange",
		"order.created",
		false,
		false,
		amqp.Publishing{
			DeliveryMode: amqp.Persistent,
			ContentType:  "application/json",
			MessageId:    order.ID,
			Body:         body,
		},
	)
	if err != nil {
		return err
	}

	log.Printf("订单消息已发送: %s", order.ID)
	return nil
}

func PublishNotification(notif model.NotificationMsg) error {
	ch := rabbitmq.GetChannel()
	defer ch.Close()

	body, err := json.Marshal(notif)
	if err != nil {
		return err
	}

	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()

	err = ch.PublishWithContext(
		ctx,
		"notification_exchange",
		"",
		false,
		false,
		amqp.Publishing{
			DeliveryMode: amqp.Persistent,
			ContentType:  "application/json",
			Body:         body,
		},
	)
	if err != nil {
		return err
	}

	log.Printf("通知消息已发送: %s", notif.OrderID)
	return nil
}
```

### 消费者：库存扣减服务

```go
package consumer

import (
	"encoding/json"
	"log"
	"time"

	"async-order/model"
	"async-order/producer"
	"async-order/rabbitmq"
)

func StartInventoryConsumer() {
	ch := rabbitmq.GetChannel()
	defer ch.Close()

	err := ch.Qos(1, 0, false)
	if err != nil {
		log.Fatalf("设置Qos失败: %v", err)
	}

	msgs, err := ch.Consume(
		"inventory_queue",
		"",
		false,
		false,
		false,
		false,
		nil,
	)
	if err != nil {
		log.Fatalf("注册库存消费者失败: %v", err)
	}

	go func() {
		for d := range msgs {
			var order model.Order
			if err := json.Unmarshal(d.Body, &order); err != nil {
				log.Printf("消息格式错误: %v", err)
				d.Ack(false)
				continue
			}

			log.Printf("扣除库存: 商品=%s 数量=%d", order.ProductID, order.Quantity)
			time.Sleep(100 * time.Millisecond)

			d.Ack(false)

			notif := model.NotificationMsg{
				OrderID:   order.ID,
				UserPhone: order.UserPhone,
				UserEmail: order.UserEmail,
				Amount:    order.Amount,
			}
			if err := producer.PublishNotification(notif); err != nil {
				log.Printf("发送通知失败: %v", err)
			}
		}
	}()

	log.Println("库存消费者已启动，等待订单...")
}
```

### 主程序

**main.go**：

```go
package main

import (
	"log"
	"os"
	"os/signal"
	"syscall"
	"time"

	"async-order/consumer"
	"async-order/producer"
	"async-order/rabbitmq"
)

func main() {
	ch := rabbitmq.GetChannel()
	if err := SetupRabbitMQ(ch); err != nil {
		log.Fatalf("初始化失败: %v", err)
	}
	ch.Close()

	mode := "consumer"
	if len(os.Args) > 1 {
		mode = os.Args[1]
	}

	if mode == "producer" {
		runProducer()
	} else {
		runConsumer()
	}
}

func runProducer() {
	order := model.Order{
		ID:        "ORD-2024-001",
		UserID:    "USR-100",
		UserPhone: "13800138000",
		UserEmail: "user@example.com",
		ProductID: "PROD-001",
		Quantity:  2,
		Amount:    198.00,
		Status:    "created",
		CreatedAt: time.Now(),
	}

	if err := producer.PublishOrderCreated(order); err != nil {
		log.Fatalf("发送失败: %v", err)
	}
	log.Println("下单完成！")
}

func runConsumer() {
	go consumer.StartInventoryConsumer()
	go consumer.StartSMSConsumer()
	go consumer.StartEmailConsumer()

	quit := make(chan os.Signal, 1)
	signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
	<-quit
	log.Println("正在关闭...")
}
```

### Kafka生产者

```go
package main

import (
	"encoding/json"
	"fmt"
	"log"
	"time"

	"github.com/IBM/sarama"
)

type ClickEvent struct {
	UserID    string `json:"user_id"`
	PageURL   string `json:"page_url"`
	Timestamp int64  `json:"timestamp"`
}

func main() {
	config := sarama.NewConfig()
	config.Producer.RequiredAcks = sarama.WaitForAll
	config.Producer.Retry.Max = 5
	config.Producer.Return.Successes = true
	config.Producer.Partitioner = sarama.NewHashPartitioner

	producer, err := sarama.NewSyncProducer([]string{"localhost:9092"}, config)
	if err != nil {
		log.Fatalf("创建生产者失败: %v", err)
	}
	defer producer.Close()

	for i := 0; i < 10; i++ {
		event := ClickEvent{
			UserID:    "user_123",
			PageURL:   "/products/" + fmt.Sprintf("%d", i),
			Timestamp: time.Now().UnixMilli(),
		}

		data, _ := json.Marshal(event)
		msg := &sarama.ProducerMessage{
			Topic: "user-clicks",
			Key:   sarama.StringEncoder(event.UserID),
			Value: sarama.ByteEncoder(data),
		}

		partition, offset, err := producer.SendMessage(msg)
		if err != nil {
			log.Printf("发送失败: %v", err)
		} else {
			log.Printf("发送成功: topic=%s partition=%d offset=%d", msg.Topic, partition, offset)
		}

		time.Sleep(500 * time.Millisecond)
	}
}
```

### Kafka消费者（消费者组）

```go
package main

import (
	"context"
	"encoding/json"
	"log"
	"os"
	"os/signal"
	"syscall"
	"time"

	"github.com/IBM/sarama"
)

type ClickEventHandler struct{}

func (h *ClickEventHandler) Setup(sarama.ConsumerGroupSession) error   { return nil }
func (h *ClickEventHandler) Cleanup(sarama.ConsumerGroupSession) error { return nil }

func (h *ClickEventHandler) ConsumeClaim(session sarama.ConsumerGroupSession, claim sarama.ConsumerGroupClaim) error {
	for msg := range claim.Messages() {
		var event ClickEvent
		json.Unmarshal(msg.Value, &event)
		log.Printf("收到点击事件: user=%s page=%s", event.UserID, event.PageURL)
		session.MarkMessage(msg, "")
	}
	return nil
}

func main() {
	config := sarama.NewConfig()
	config.Consumer.Group.Rebalance.Strategy = sarama.NewBalanceStrategyRoundRobin()
	config.Consumer.Offsets.Initial = sarama.OffsetOldest
	config.Consumer.Offsets.AutoCommit.Enable = false

	group, err := sarama.NewConsumerGroup(
		[]string{"localhost:9092"},
		"click-analytics-group",
		config,
	)
	if err != nil {
		log.Fatalf("创建消费者组失败: %v", err)
	}
	defer group.Close()

	ctx, cancel := context.WithCancel(context.Background())
	defer cancel()

	handler := &ClickEventHandler{}

	go func() {
		for {
			err := group.Consume(ctx, []string{"user-clicks"}, handler)
			if err != nil {
				log.Printf("消费错误: %v", err)
			}
		}
	}()

	log.Println("消费者组已启动，等待消息...")

	sigChan := make(chan os.Signal, 1)
	signal.Notify(sigChan, syscall.SIGINT, syscall.SIGTERM)
	<-sigChan
	cancel()
	log.Println("正在关闭...")
}
```

## 90.3 实战：异步下单系统设计

现在把RabbitMQ和Kafka结合起来，构建一个完整的异步下单系统。

### 系统架构

```
                         ┌─────────────────────────────────────────┐
                         │              消息流                       │
                         │                                          │
用户点击下单              │                                          │
    │                    │                                          │
    ↓                    │                                          │
 订单服务 ───────────────→ RabbitMQ ──→ 库存服务（扣库存）           │
    │                    │   │            │                         │
    │                    │   │     ┌──────┘                         │
    │                    │   │     │ 扣库存成功                      │
    │                    │   │     ↓                                │
    │                    │   │  RabbitMQ ──→ 短信服务               │
    │                    │   │              ──→ 邮件服务             │
    │                    │   │                                     │
    │                    │                                         │
    │                    ├──→ Kafka ──→ 实时分析（用户行为）         │
    │                    │           ──→ 数据仓库（离线报表）         │
    │                    │                                         │
    └─ 返回"下单成功" ───┘                                         │
```

### 完整下单流程

```
1. 用户POST /orders

2. 订单服务（Producer）：
   a. 验证请求参数
   b. 创建订单记录（DB，status=pending）
   c. 发送order.created到RabbitMQ（topic: order_exchange, routing_key: order.created）
   d. 发送用户行为事件到Kafka（topic: user-behaviors, key: user_id）
   e. 返回HTTP 201 "下单成功，订单号XXX"

3. 库存服务（RabbitMQ Consumer）：
   a. 收到order.created消息
   b. 扣减库存（Redis/DB）
   c. 如果扣减成功 → 发送通知到notification_exchange
   d. 如果扣减失败（库存不足） → 更新订单状态为failed → 发送退款消息
   e. ACK

4. 短信服务（RabbitMQ Consumer）：
   a. 收到通知消息
   b. 调用短信平台API发送短信
   c. ACK（即使短信发送失败也ACK——短信不是核心交易）

5. 邮件服务（RabbitMQ Consumer）：
   a. 收到通知消息
   b. 调用邮件服务发送邮件
   c. ACK

6. 实时分析服务（Kafka Consumer Group: analytics）：
   a. 消费用户行为事件
   b. 实时统计：热门商品、转化率等

7. 数据仓库服务（Kafka Consumer Group: warehouse）：
   a. 消费同样的数据
   b. 写入HDFS/ClickHouse，供离线分析
```

### 关键设计决策

| 决策点 | 选择 | 原因 |
|--------|------|------|
| 为什么库存扣选用RabbitMQ | 需要ACK确认，保证不丢 | 库存扣减不能丢，也不能重复扣 |
| 为什么通知用RabbitMQ fanout | 多个消费者都要收到 | 短信和邮件各自独立 |
| 为什么行为事件用Kafka | 海量+可回溯 | 每天百万级点击，分析师可能需要回放 |
| 为什么Kafka消费者分两个组 | 实时和离线不同节奏 | 实时要快，离线可以慢 |
| 为什么订单状态回写DB | 幂等性保障 | 即使消息重复，查DB状态做幂等判断 |

### 核心代码：订单服务

```go
func CreateOrder(w http.ResponseWriter, r *http.Request) {
	var req CreateOrderRequest
	json.NewDecoder(r.Body).Decode(&req)

	order := model.Order{
		ID:        generateOrderID(),
		UserID:    req.UserID,
		ProductID: req.ProductID,
		Quantity:  req.Quantity,
		Status:    "pending",
		CreatedAt: time.Now(),
	}

	err := db.Create(&order).Error
	if err != nil {
		http.Error(w, "创建订单失败", 500)
		return
	}

	go publishToRabbitMQ(order)
	go publishToKafka(order)

	w.Header().Set("Content-Type", "application/json")
	w.WriteHeader(http.StatusCreated)
	json.NewEncoder(w).Encode(map[string]string{
		"order_id": order.ID,
		"message":  "下单成功",
	})
}
```

注意：`publishToRabbitMQ` 和 `publishToKafka` 用 `go` 关键字异步执行——即使MQ暂时不可用也不阻塞用户，但需要在内部做重试和失败处理。

🤔 **想多一点**：消息队列引入的新问题

消息队列解决了"同步阻塞"的问题，但引入了新的挑战——**数据一致性**。在异步下单系统中：

- 用户收到"下单成功" → 但库存扣减可能失败 → 用户以为下单成功了，实际上库存不够

这是分布式系统的经典难题：**最终一致性**。解决方案不是"让消息队列变同步"，而是：

1. **明确告知用户**："下单成功"≠"发货成功"。"您的订单已提交，我们正在准备中"
2. **补偿机制**：库存扣减失败 → 自动退款 → 发通知告诉用户
3. **状态追踪**：用户随时可以查"我的订单现在到哪一步了"
4. **幂等性设计**：每一步都可以安全重试
