## 我在做什么？

计算机里的时间不像墙上的钟那么简单。程序需要知道"现在几点"，需要计算两个时间点之间差了多少秒，需要把时间戳转换成人类可读的"2026年6月7日 15:30"。

C语言通过 `<time.h>` 头文件提供了一套时间处理函数。核心概念是**时间戳**——一个简单的整数，代表从某个固定起点（Epoch）到现在经过的秒数。

想象时间是一条无限长的数轴。1970年1月1日 00:00:00 UTC 是数轴上的原点（0）。每过一秒，数字就加 1。2026年的某个时刻，这个数字大约是 17.8 亿。这个数字就是 `time_t` 时间戳。

`*此术语见附录F*`：时间戳、Unix时间、Epoch。

## 52.1 time_t：时间戳

`time_t` 是一个整数类型（通常是 `long` 或 `long long`），存储从 **Epoch（1970-01-01 00:00:00 UTC）** 到现在的秒数。

```
时间轴:
  ----|----|----|----|----|----|----|----|---->
      1970             2000           2026
      (0)           (946684800)    (约1780000000)

  time_t 就是一个整数，表示从 1970 年 1 月 1 日起的秒数
```

```c
#include <stdio.h>
#include <time.h>

int main(void) {
    time_t now;
    time(&now);  /* 获取当前时间戳，存入 now */

    printf("当前时间戳: %ld\n", (long)now);
    printf("（从 1970-01-01 00:00:00 UTC 起经过的秒数）\n");

    return 0;
}
```

运行输出：
```
当前时间戳: 1780000000
（从 1970-01-01 00:00:00 UTC 起经过的秒数）
```

### 获取时间戳的三种方式

```c
time_t t1 = time(NULL);      /* 方式1: 返回时间戳 */

time_t t2;
time(&t2);                   /* 方式2: 通过指针写入 */

time_t t3 = 0;               /* 方式3: 先初始化再调用 */
time(&t3);
```

三种方式效果相同，`time(NULL)` 最简洁。

## 52.2 struct tm：分解时间

时间戳是一个数字，不适于人类阅读。`struct tm` 把时间戳分解为年、月、日、时、分、秒等字段：

```c
struct tm {
    int tm_sec;    /* 秒，0-59（偶尔有 60 表示闰秒） */
    int tm_min;    /* 分，0-59 */
    int tm_hour;   /* 时，0-23 */
    int tm_mday;   /* 日，1-31 */
    int tm_mon;    /* 月，0-11（0 = 一月！注意偏移） */
    int tm_year;   /* 年，从 1900 起算（2026年 = 126） */
    int tm_wday;   /* 星期几，0-6（0 = 周日） */
    int tm_yday;   /* 一年中的第几天，0-365 */
    int tm_isdst;  /* 夏令时标志：正数=是，0=否，负数=未知 */
};
```

```
时间戳 (time_t)                    分解时间 (struct tm)
  1780000000          ──────>      年=2026, 月=6, 日=7
                    localtime()    时=15, 分=30, 秒=0
                                   星期几=0(周日)

分解时间 (struct tm)               时间戳 (time_t)
  年=2026, 月=6, 日=7   ──────>      1780000000
  时=15, 分=30, 秒=0    mktime()
```

### localtime 和 gmtime

```c
#include <stdio.h>
#include <time.h>

int main(void) {
    time_t now;
    time(&now);

    /* localtime: 转为本地时区时间 */
    struct tm *local = localtime(&now);
    printf("本地时间: %d-%02d-%02d %02d:%02d:%02d\n",
           local->tm_year + 1900,  /* 年份要加 1900 */
           local->tm_mon + 1,       /* 月份要加 1 */
           local->tm_mday,
           local->tm_hour,
           local->tm_min,
           local->tm_sec);

    /* gmtime: 转为 UTC 时间 */
    struct tm *utc = gmtime(&now);
    printf("UTC 时间: %d-%02d-%02d %02d:%02d:%02d\n",
           utc->tm_year + 1900,
           utc->tm_mon + 1,
           utc->tm_mday,
           utc->tm_hour,
           utc->tm_min,
           utc->tm_sec);

    return 0;
}
```

**重要警告**：`localtime()` 和 `gmtime()` 返回的是**静态内部缓冲区的指针**。每次调用会覆盖上次的结果。如果需要保留多个时间，用 `memcpy` 复制，或者使用线程安全版本 `localtime_r()` / `gmtime_r()`。

```c
/* ❌ 危险：两次调用 localtime，t1 和 t2 指向同一块内存 */
struct tm *t1 = localtime(&time1);
struct tm *t2 = localtime(&time2);
/* 此时 t1 和 t2 都指向 time2 的分解结果！ */

/* ✅ 正确：使用 localtime_r（线程安全版） */
struct tm t1, t2;
localtime_r(&time1, &t1);
localtime_r(&time2, &t2);
```

## 52.3 mktime：分解时间转回时间戳

`mktime()` 把 `struct tm` 转回 `time_t`。它会自动修正不合法的日期（比如"1月32日"自动变成"2月1日"）。

```c
#include <stdio.h>
#include <time.h>

int main(void) {
    struct tm date = {0};

    /* 设置 2026-06-07 12:00:00 */
    date.tm_year = 2026 - 1900;  /* 年份减 1900 */
    date.tm_mon  = 6 - 1;         /* 月份减 1 */
    date.tm_mday = 7;
    date.tm_hour = 12;
    date.tm_min  = 0;
    date.tm_sec  = 0;
    date.tm_isdst = -1;           /* 让系统自动判断夏令时 */

    time_t timestamp = mktime(&date);
    if (timestamp == -1) {
        printf("转换失败\n");
        return 1;
    }

    printf("2026-06-07 12:00:00 的时间戳 = %ld\n", (long)timestamp);

    /* mktime 还会自动填充 tm_wday 和 tm_yday */
    printf("星期几: %d (0=周日)\n", date.tm_wday);
    printf("一年中的第 %d 天\n", date.tm_yday + 1);

    return 0;
}
```

## 52.4 strftime：格式化时间输出

`strftime()` 是时间格式化利器，用格式字符串控制输出：

```c
#include <stdio.h>
#include <time.h>

int main(void) {
    time_t now;
    time(&now);
    struct tm *local = localtime(&now);

    char buf[128];

    /* 各种格式示例 */
    strftime(buf, sizeof(buf), "%Y-%m-%d %H:%M:%S", local);
    printf("ISO 格式:    %s\n", buf);

    strftime(buf, sizeof(buf), "%Y年%m月%d日 %H:%M:%S", local);
    printf("中文格式:    %s\n", buf);

    strftime(buf, sizeof(buf), "%A, %B %d, %Y", local);
    printf("英文格式:    %s\n", buf);

    strftime(buf, sizeof(buf), "%c", local);
    printf("本地格式:    %s\n", buf);

    strftime(buf, sizeof(buf), "%F %T", local);
    printf("便捷格式:    %s\n", buf);

    return 0;
}
```

### 常用 strftime 格式符

| 格式符 | 含义 | 示例 |
|--------|------|------|
| `%Y` | 四位年份 | 2026 |
| `%y` | 两位年份 | 26 |
| `%m` | 月份（01-12） | 06 |
| `%d` | 日期（01-31） | 07 |
| `%H` | 24小时制小时（00-23） | 15 |
| `%I` | 12小时制小时（01-12） | 03 |
| `%M` | 分钟（00-59） | 30 |
| `%S` | 秒（00-59） | 45 |
| `%p` | AM/PM | PM |
| `%A` | 完整星期名 | Sunday |
| `%a` | 缩写星期名 | Sun |
| `%B` | 完整月份名 | June |
| `%b` | 缩写月份名 | Jun |
| `%c` | 本地日期时间表示 | Sun Jun 7 15:30:45 2026 |
| `%F` | `%Y-%m-%d` 的简写 | 2026-06-07 |
| `%T` | `%H:%M:%S` 的简写 | 15:30:45 |

## 52.5 difftime：计算时间差

`difftime()` 返回两个 `time_t` 之间的秒数差（`double` 类型）：

```c
#include <stdio.h>
#include <time.h>

int main(void) {
    time_t start, end;

    time(&start);
    printf("开始计时...\n");

    /* 模拟耗时操作 */
    printf("（按回车键结束计时）");
    getchar();

    time(&end);

    double elapsed = difftime(end, start);
    printf("经过了 %.0f 秒\n", elapsed);

    return 0;
}
```

### 计算两个日期之间的天数

```c
#include <stdio.h>
#include <time.h>

int days_between(int y1, int m1, int d1, int y2, int m2, int d2) {
    struct tm date1 = {0};
    struct tm date2 = {0};

    date1.tm_year = y1 - 1900;
    date1.tm_mon  = m1 - 1;
    date1.tm_mday = d1;
    date1.tm_isdst = -1;

    date2.tm_year = y2 - 1900;
    date2.tm_mon  = m2 - 1;
    date2.tm_mday = d2;
    date2.tm_isdst = -1;

    time_t t1 = mktime(&date1);
    time_t t2 = mktime(&date2);

    if (t1 == -1 || t2 == -1) return -1;

    double diff = difftime(t2, t1);
    return (int)(diff / (60 * 60 * 24));  /* 秒转天 */
}

int main(void) {
    int days = days_between(2026, 1, 1, 2026, 6, 7);
    printf("2026-01-01 到 2026-06-07 有 %d 天\n", days);

    days = days_between(2026, 6, 7, 2027, 1, 1);
    printf("2026-06-07 到 2027-01-01 有 %d 天\n", days);

    return 0;
}
```

## 52.6 clock：程序运行时间（CPU 时间）

`clock()` 返回程序从启动到现在的 CPU 时钟滴答数，配合 `CLOCKS_PER_SEC` 可以计算 CPU 时间：

```c
#include <stdio.h>
#include <time.h>

int main(void) {
    clock_t start = clock();

    /* 做一些计算 */
    long sum = 0;
    for (long i = 0; i < 100000000; i++) {
        sum += i;
    }

    clock_t end = clock();

    double cpu_time = (double)(end - start) / CLOCKS_PER_SEC;
    printf("计算耗时: %.3f 秒 (CPU时间)\n", cpu_time);
    printf("结果: %ld\n", sum);

    return 0;
}
```

**注意**：`clock()` 测量的是 CPU 时间（进程实际占用 CPU 的时间），不是挂钟时间。如果程序在等待 I/O 或休眠，这段时间不计入 CPU 时间。

## 52.7 完整代码清单

```c
/*
 * time_demo.c —— C语言时间处理完整演示
 * 编译: gcc -std=c99 -Wall -o time_demo time_demo.c
 */

#include <stdio.h>
#include <time.h>
#include <string.h>

/* 打印 tm 结构体的所有字段 */
void print_tm(const char *label, const struct tm *t) {
    printf("%s:\n", label);
    printf("  年: %d\n",   t->tm_year + 1900);
    printf("  月: %d\n",   t->tm_mon + 1);
    printf("  日: %d\n",   t->tm_mday);
    printf("  时: %d\n",   t->tm_hour);
    printf("  分: %d\n",   t->tm_min);
    printf("  秒: %d\n",   t->tm_sec);
    printf("  星期: %d (0=周日)\n", t->tm_wday);
    printf("  年第几天: %d\n", t->tm_yday + 1);
    printf("  夏令时: %d\n", t->tm_isdst);
}

/* 计算两个日期相差的天数 */
int days_between(int y1, int m1, int d1, int y2, int m2, int d2) {
    struct tm date1 = {0};
    struct tm date2 = {0};

    date1.tm_year = y1 - 1900;
    date1.tm_mon  = m1 - 1;
    date1.tm_mday = d1;
    date1.tm_isdst = -1;

    date2.tm_year = y2 - 1900;
    date2.tm_mon  = m2 - 1;
    date2.tm_mday = d2;
    date2.tm_isdst = -1;

    time_t t1 = mktime(&date1);
    time_t t2 = mktime(&date2);

    if (t1 == -1 || t2 == -1) return -1;

    return (int)(difftime(t2, t1) / 86400);
}

int main(void) {
    /* 1. 获取当前时间戳 */
    printf("===== 当前时间戳 =====\n");
    time_t now;
    time(&now);
    printf("time_t = %ld\n\n", (long)now);

    /* 2. localtime 和 gmtime 对比 */
    printf("===== localtime vs gmtime =====\n");
    struct tm local_tm, utc_tm;
    localtime_r(&now, &local_tm);
    gmtime_r(&now, &utc_tm);
    print_tm("本地时间", &local_tm);
    printf("\n");
    print_tm("UTC 时间", &utc_tm);

    /* 3. strftime 格式化 */
    printf("\n===== strftime 格式化 =====\n");
    char buf[128];
    const char *formats[] = {
        "%Y-%m-%d %H:%M:%S",
        "%Y年%m月%d日 %H:%M:%S",
        "%A, %B %d, %Y",
        "%c",
        "%F %T",
    };
    for (int i = 0; i < 5; i++) {
        strftime(buf, sizeof(buf), formats[i], &local_tm);
        printf("  \"%s\"  ->  %s\n", formats[i], buf);
    }

    /* 4. mktime：构造指定时间 */
    printf("\n===== mktime 构造时间 =====\n");
    struct tm custom = {0};
    custom.tm_year = 2026 - 1900;
    custom.tm_mon  = 6 - 1;
    custom.tm_mday = 7;
    custom.tm_hour = 12;
    custom.tm_min  = 0;
    custom.tm_sec  = 0;
    custom.tm_isdst = -1;

    time_t custom_ts = mktime(&custom);
    strftime(buf, sizeof(buf), "%F %T", &custom);
    printf("构造的时间: %s, 时间戳: %ld\n", buf, (long)custom_ts);

    /* 5. difftime：时间差 */
    printf("\n===== difftime 计算日期差 =====\n");
    int diff = days_between(2026, 1, 1, 2026, 6, 7);
    printf("2026-01-01 到 2026-06-07: %d 天\n", diff);

    diff = days_between(1970, 1, 1, 2026, 6, 7);
    printf("1970-01-01 到 2026-06-07: %d 天\n", diff);

    /* 6. clock：CPU 时间 */
    printf("\n===== clock CPU 时间 =====\n");
    clock_t start = clock();
    volatile long sum = 0;
    for (long i = 0; i < 50000000; i++) {
        sum += i;
    }
    clock_t end_c = clock();
    double cpu_sec = (double)(end_c - start) / CLOCKS_PER_SEC;
    printf("5000万次累加耗时: %.3f 秒 (CPU时间)\n", cpu_sec);

    return 0;
}
```

## 我做得对不对？

1. 运行程序，看到当前时间戳是一个 10 位整数。
2. `localtime` 输出和系统时钟一致，`gmtime` 输出 UTC 时间（北京时间 = UTC+8）。
3. `strftime` 各种格式符都能正确输出。
4. `mktime` 构造的 2026-06-07 12:00:00 时间戳正确。
5. `days_between` 计算的天数可以手动验证。

---

## 不对怎么办？

### 常见Bug 1：忘记 tm_year += 1900 和 tm_mon += 1

```c
/* ❌ 直接打印 tm_year，输出 126 而不是 2026 */
struct tm *t = localtime(&now);
printf("%d\n", t->tm_year);  /* 输出 126 */

/* ❌ 直接打印 tm_mon，输出 5 而不是 6 */
printf("%d\n", t->tm_mon);   /* 输出 5 */
```

✅ `tm_year` 要加 1900，`tm_mon` 要加 1。

### 常见Bug 2：localtime 返回指针被覆盖

```c
/* ❌ t1 和 t2 指向同一块静态内存 */
struct tm *t1 = localtime(&time1);
struct tm *t2 = localtime(&time2);
/* t1 的内容被第二次 localtime 调用覆盖了！ */
```

✅ 使用 `localtime_r()` 或立即 `memcpy` 复制。

### 常见Bug 3：difftime 和 time_t 相减混淆

```c
/* ❌ 直接用 time_t 相减，结果可能溢出或类型不匹配 */
time_t diff = end - start;

/* ✅ 使用 difftime，返回 double */
double diff = difftime(end, start);
```

✅ 始终使用 `difftime()`，不要直接相减 `time_t`。

### 常见Bug 4：strftime 缓冲区太小

```c
/* ❌ 缓冲区只有 10 字节，某些格式会溢出 */
char buf[10];
strftime(buf, sizeof(buf), "%A, %B %d, %Y", &tm);  /* 可能截断 */
```

✅ 给缓冲区至少 64 字节，或先用 `strftime(NULL, 0, fmt, &tm)` 获取所需大小。

### 常见Bug 5：使用 mktime 前未初始化 struct tm

```c
/* ❌ struct tm 未初始化，垃圾值影响 mktime 结果 */
struct tm date;
date.tm_year = 2026 - 1900;
date.tm_mon = 6 - 1;
date.tm_mday = 7;
/* tm_hour, tm_min, tm_sec 都是垃圾值！ */
time_t t = mktime(&date);
```

✅ 使用 `struct tm date = {0};` 零初始化，或显式设置所有字段。

## 本章小结

| 要点 | 说明 |
|------|------|
| time_t | 时间戳类型，从 1970-01-01 00:00:00 UTC 起的秒数 |
| Epoch | Unix 时间起点：1970年1月1日 00:00:00 UTC |
| time() | 获取当前时间戳 |
| struct tm | 分解时间，包含年月日时分秒等字段 |
| tm_year | 从 1900 年起算，打印时需 +1900 |
| tm_mon | 0-11，打印时需 +1 |
| localtime() | 时间戳转本地时间（返回静态指针，注意覆盖） |
| gmtime() | 时间戳转 UTC 时间 |
| mktime() | 分解时间转时间戳，自动修正非法日期 |
| strftime() | 格式化时间输出，支持多种格式符 |
| difftime() | 计算两个时间戳的秒数差 |
| clock() | 获取 CPU 时钟滴答数，用于性能测量 |
